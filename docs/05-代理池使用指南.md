# 代理池使用指南 - agent-browser 集成

## 概述

本文档说明如何在 agent-browser 中使用 WIPO 项目的代理池。

### 代理池信息

| 配置项 | 值 |
|--------|-----|
| 代理服务器 | `23.148.244.2` |
| 端口范围 | `10000-10252` |
| 代理数量 | 253个 |
| 协议 | SOCKS5 |
| 白名单 | 仅允许 `216.116.160.78、216.116.160.79` |

### 端口映射规则

端口与IP的映射关系：
```
端口 = 10000 + (IP末位 - 2)
```

示例：
- `23.148.244.2` → 端口 `10000`
- `23.148.244.100` → 端口 `10098`
- `23.148.244.254` → 端口 `10252`

## 代理轮换使用

### 基本轮换示例

```javascript
const { ProxyManager } = require('./src/utils/proxy-manager');

// 初始化代理管理器
const proxyManager = new ProxyManager({
  proxyHost: '23.148.244.2',
  portStart: 10000,
  portEnd: 10252,
  strategy: 'round-robin',  // 轮换策略
  cooldownMs: 30000         // 30秒冷却
});

// 获取下一个代理
const proxy = proxyManager.getNextProxy();
console.log(proxy);
// 输出: { url: 'socks5://23.148.244.2:10001', index: 1, port: 10001, ip: '23.148.244.3' }
```

### 与 agent-browser 集成

```javascript
const { AgentBrowser } = require('agent-browser');
const { ProxyManager } = require('./src/utils/proxy-manager');

const proxyManager = new ProxyManager();

async function browseWithProxy(url) {
  // 获取代理
  const proxy = proxyManager.getNextProxy();
  
  try {
    const browser = new AgentBrowser({
      proxy: proxy.url
    });
    
    const result = await browser.browse(url);
    
    // 标记成功
    proxyManager.markSuccess(proxy.index);
    
    return result;
  } catch (error) {
    // 标记失败
    proxyManager.markFailed(proxy.index);
    throw error;
  }
}
```

### 并行请求示例

```javascript
const { ProxyManager } = require('./src/utils/proxy-manager');

const proxyManager = new ProxyManager();

async function parallelBrowse(urls) {
  // 获取多个代理
  const proxies = proxyManager.getProxiesForParallel(urls.length);
  
  const promises = urls.map((url, i) => {
    const browser = new AgentBrowser({
      proxy: proxies[i].url
    });
    return browser.browse(url);
  });
  
  return Promise.all(promises);
}
```

## 轮换策略

ProxyManager 支持三种轮换策略：

### 1. Round-Robin（默认）

按顺序轮换使用代理，适合均匀分布请求。

```javascript
const proxyManager = new ProxyManager({
  strategy: 'round-robin'
});
```

### 2. Random

随机选择代理。

```javascript
const proxyManager = new ProxyManager({
  strategy: 'random'
});
```

### 3. Least-Used

选择使用次数最少的代理，适合长期运行的任务。

```javascript
const proxyManager = new ProxyManager({
  strategy: 'least-used'
});
```

## 完整示例

### 示例1：批量爬取

```javascript
const { AgentBrowser } = require('agent-browser');
const { ProxyManager } = require('./src/utils/proxy-manager');

async function batchCrawl(urls) {
  const proxyManager = new ProxyManager({
    strategy: 'round-robin',
    cooldownMs: 30000  // 同一代理30秒内不重复使用
  });
  
  const results = [];
  
  for (const url of urls) {
    const proxy = proxyManager.getNextProxy();
    
    try {
      const browser = new AgentBrowser({
        proxy: proxy.url,
        timeout: 60000
      });
      
      console.log(`[${proxy.ip}] Crawling: ${url}`);
      
      const result = await browser.browse(url);
      proxyManager.markSuccess(proxy.index);
      
      results.push({ url, success: true, data: result });
    } catch (error) {
      proxyManager.markFailed(proxy.index);
      results.push({ url, success: false, error: error.message });
    }
    
    // 请求间隔
    await new Promise(r => setTimeout(r, 2000));
  }
  
  // 打印统计
  console.log(proxyManager.getStats());
  
  return results;
}

// 使用
batchCrawl([
  'https://www3.wipo.int/branddb/en/?p=search%3Bq=BRAND1',
  'https://www3.wipo.int/branddb/en/?p=search%3Bq=BRAND2',
  // ...
]);
```

### 示例2：使用 socks-proxy-agent

```javascript
const { SocksProxyAgent } = require('socks-proxy-agent');
const { ProxyManager } = require('./src/utils/proxy-manager');

const proxyManager = new ProxyManager();

function getProxyAgent() {
  const proxy = proxyManager.getNextProxy();
  return {
    agent: new SocksProxyAgent(proxy.url),
    proxyInfo: proxy
  };
}

// 用于 fetch 或 axios
async function fetchWithProxy(url) {
  const { agent, proxyInfo } = getProxyAgent();
  
  const response = await fetch(url, {
    agent,
    headers: {
      'User-Agent': 'Mozilla/5.0 ...'
    }
  });
  
  return response;
}
```

## 故障排查

### 连接超时

**症状：** 代理连接超时

**可能原因：**
1. IP不在白名单中
2. 代理服务未运行
3. 网络问题

**解决方法：**
```bash
# 1. 检查当前服务器IP
curl -s https://api.ipify.org/

# 2. 测试代理连接
curl -v --socks5 23.148.244.2:10000 https://api.ipify.org/

# 3. 检查代理服务状态（需要SSH访问）
ssh -p 30022 root@23.148.244.2 "systemctl status 3proxy"
```

### IP白名单配置

代理服务器只允许白名单IP访问。如需添加新IP：

```bash
# SSH到代理服务器
ssh -p 30022 root@23.148.244.2

# 编辑配置文件
vim /etc/3proxy/3proxy.cfg

# 在 allow 行添加新IP
allow user,216.116.160.78,NEW_IP_HERE

# 重启服务
systemctl restart 3proxy
```

### 代理被封锁

**症状：** 目标网站返回403/429

**解决方法：**
```javascript
// 增加冷却时间
const proxyManager = new ProxyManager({
  cooldownMs: 60000  // 60秒
});

// 或使用 least-used 策略
const proxyManager = new ProxyManager({
  strategy: 'least-used'
});
```

## 最佳实践

### 1. 冷却时间

设置适当的冷却时间，避免同一IP频繁请求：

```javascript
const proxyManager = new ProxyManager({
  cooldownMs: 30000  // 推荐30-60秒
});
```

### 2. 错误处理

正确处理代理失败，自动切换：

```javascript
async function browseWithRetry(url, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    const proxy = proxyManager.getNextProxy();
    try {
      const result = await browse(url, proxy);
      proxyManager.markSuccess(proxy.index);
      return result;
    } catch (error) {
      proxyManager.markFailed(proxy.index);
      console.log(`Proxy ${proxy.ip} failed, trying next...`);
    }
  }
  throw new Error('All proxies failed');
}
```

### 3. 监控统计

定期检查代理使用统计：

```javascript
// 每100个请求打印一次统计
if (requestCount % 100 === 0) {
  console.log(proxyManager.getStats());
}
```

### 4. 状态持久化

ProxyManager 会自动保存状态到 `output/proxy-state.json`，重启后恢复。
